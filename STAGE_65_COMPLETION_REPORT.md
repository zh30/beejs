// 智能分发策略
if data.len() < 1024 {
    // 小脚本 -> L1 零拷贝
    self.l1_cache.put(cache_key, data).await;
} else if data.len() < 1024 * 1024 {
    // 中等脚本 -> L2 智能
    let mut l2 = self.l2_cache.write().unwrap();
    l2.put(cache_key, data);
} else {
    // 大脚本 -> L3 内存映射

}
